import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns

# Load the dataset from an online source
url = 'https://raw.githubusercontent.com/fivethirtyeight/data/master/bad-drivers/bad-drivers.csv'
df = pd.read_csv(url)

# Display the first few rows of the dataset
print(df.head())

# Display the columns in the dataset
print(df.columns)

# Rename columns for easier access
df.rename(columns={'Number of drivers involved in fatal collisions per billion miles': 'NumDrivers',
'Percentage Of Drivers Involved In Fatal Collisions Who Were Speeding': 'PctSpeeding',
'Percentage Of Drivers Involved In Fatal Collisions Who Were Alcohol-Impaired': 'PctAlcohol',
'Percentage Of Drivers Involved In Fatal Collisions Who Were Not Distracted': 'PctNotDistracted',
'Percentage Of Drivers Involved In Fatal Collisions Who Had Not Been Involved In Any Previous Accidents': 'PctNoPrevAccidents',
'Car Insurance Premiums ($)': 'InsurancePremiums',
'Losses incurred by insurance companies for collisions per insured driver': 'Losses'}, inplace=True)

# Display the updated DataFrame
print(df.head())

# Exploratory Data Analysis (EDA)

# Plot the number of drivers involved
plt.figure(figsize=(12, 6))
sns.histplot(data=df, x='NumDrivers', kde=True, palette='viridis')
plt.title('Distribution of Number of Drivers Involved in Fatal Collisions per Billion Miles')
plt.xlabel('Number of Drivers Involved')
plt.ylabel('Frequency')
plt.grid(True) # Adding grid lines
plt.show()

# Plot the percentage of drivers with previous incidents
plt.figure(figsize=(12, 6))
sns.histplot(data=df, x='PctNoPrevAccidents', kde=True, palette='viridis')
plt.title('Distribution of Percentage of Drivers with No Previous Accidents')
plt.xlabel('Percentage of Drivers with No Previous Accidents')
plt.ylabel('Frequency')
plt.grid(True) # Adding grid lines
plt.show()
